FeatureValue
ExtrapolateObjPositionExtractor::extract(
    const Context& context,
    FeatureState&) const
{
auto& obj = context.simulationObject();

double dt =
    context.currentSystemTime() -
    context.objectTime();

double newX =
    obj.x_pos_ +
    obj.x_vel_ * dt +
    0.5 * obj.x_accel_ * dt * dt;

double newY =
    obj.y_pos_ +
    obj.y_vel_ * dt +
    0.5 * obj.y_accel_ * dt * dt;

double newZ =
    obj.z_pos_ +
    obj.z_vel_ * dt +
    0.5 * obj.z_accel_ * dt * dt;

return FeatureValue({
    newX,
    newY,
    newZ
});
